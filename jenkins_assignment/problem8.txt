import groovy.json.JsonSlurper;
pipeline {
    agent any 
    stages {
        stage('Build') { 
            steps {
                script {
                        def jsonData = new JsonSlurper();
                        def getTodos = new URL("https://jsonplaceholder.typicode.com/todos").openConnection();
                        def getResCode = getTodos.getResponseCode();
                        
                        if (getResCode.equals(200)){  
                            data = getTodos.getInputStream().getText()
                            todosJsonData = jsonData.parseText(data)   
                        
                        for( item in todosJsonData){
                            if(item.completed==false)
                                println(item)
                        }
                          
                    }
                    
                }
            }
        }

    }
}
